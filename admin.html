Supabase 데이터베이스 테이블 셋업 가이드

광고 및 회원 등급 관리에 필요한 Supabase SQL 셋업 명령어 스크립트입니다.

아래의 SQL 구문을 전체 복사하여 **Supabase Dashboard -> SQL Editor -> New Query** 창에 붙여넣고 **Run** 버튼을 눌러 실행해주시기 바랍니다. 

-- 0. 고객 프로필 테이블 membership_tier 컬럼 추가
ALTER TABLE public.user_profiles ADD COLUMN IF NOT EXISTS membership_tier VARCHAR(50) DEFAULT 'regular';
ALTER TABLE public.user_profiles ADD COLUMN IF NOT EXISTS membership_expires_at TIMESTAMP WITH TIME ZONE;

-- 0-1. 모의입찰(mock_bids) 테이블 구축
CREATE TABLE IF NOT EXISTS public.mock_bids (
    property_id BIGINT REFERENCES public.properties(id) ON DELETE CASCADE,
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    bid_price BIGINT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (property_id, user_id)
);

-- 0-2. 전문가 상담(expert_consultations) 테이블 구축
CREATE TABLE IF NOT EXISTS public.expert_consultations (
    id SERIAL PRIMARY KEY,
    property_id BIGINT REFERENCES public.properties(id) ON DELETE CASCADE,
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    expert_name VARCHAR(255) NOT NULL,
    status VARCHAR(50) DEFAULT 'pending',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 1. 광고판(ads) 데이터베이스 테이블 구축
CREATE TABLE IF NOT EXISTS public.ads (
    id SERIAL PRIMARY KEY,
    slot_name VARCHAR(255) UNIQUE NOT NULL,
    active BOOLEAN DEFAULT TRUE,
    type VARCHAR(50) DEFAULT 'direct',
    title VARCHAR(255),
    desc TEXT,
    ad_code TEXT,
    image_url TEXT,
    link_url TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 3. 고객 프로필 회원 등급(user_profiles) 테이블 구축
CREATE TABLE IF NOT EXISTS public.user_profiles (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    email VARCHAR(255) NOT NULL,
    grade VARCHAR(10) DEFAULT 'C',
    upgrade_requested BOOLEAN DEFAULT FALSE,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 기존 테이블이 존재할 경우 upgrade_requested 컬럼 추가 SQL
ALTER TABLE public.user_profiles ADD COLUMN IF NOT EXISTS upgrade_requested BOOLEAN DEFAULT FALSE;

-- 신규 유저 가입 시 자동으로 등급 테이블 행을 연계 생성하는 DB 트리거 함수 작성
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.user_profiles (id, email, grade, upgrade_requested)
  VALUES (new.id, new.email, 'C', false)
  ON CONFLICT (id) DO NOTHING;
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 트리거 생성
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
